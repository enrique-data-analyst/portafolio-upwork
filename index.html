import pandas as pd
import numpy as np

# Load and clean dataset for business reporting
def process_raw_data(file_path):
    df = pd.read_csv(file_path)
    
    # Handling outliers and cleaning currency formats
    df['revenue'] = df['revenue'].replace('[\$,]', '', regex=True).astype(float)
    df_clean = df.dropna(subset=['customer_id'])
    
    # Returning aggregated business insights
    return df_clean.groupby('region').agg({'revenue': 'sum'})

print("Process completed: Data ready for visualization.")
